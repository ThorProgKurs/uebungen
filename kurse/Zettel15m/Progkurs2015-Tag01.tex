\begin{aufg}
Was machen folgende Programme (kein C-Code)?

\begin{algorithm}[H]
\caption{}
\algsetup{indent=1.5em}
\begin{algorithmic}[1]
\REQUIRE Ganze Zahl $c\in\mathbb{N}$
\ENSURE Entweder \verb|Ja| oder \verb|Nein|.
\STATE \SET $n := 2$.
\IF{$n>\sqrt{c}$} \label{1Start}
\STATE Gebe \verb|Ja| aus.
\RETURN $0$
\ENDIF
\IF{$n$ teilt $c$}
\STATE Gebe \verb|Nein| aus.
\RETURN $0$
\ENDIF
\STATE \SET $n := n + 1$
\STATE \GOTO \ref{1Start}
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
\caption{}
\algsetup{indent=1.5em}
\begin{algorithmic}[1]
\REQUIRE Ganze Zahlen $a,b\in\mathbb{N}$
\ENSURE Eine ganze Zahl $k\in\mathbb{N}$
\IF{$a=0$} 
\STATE Gebe $b$ aus.
\RETURN $0$
\ENDIF
\IF{$b=0$} \label{3Start}
\STATE Gebe $a$ aus.
\RETURN $0$
\ENDIF
\IF{$a>b$}
\STATE \SET $a = a - b$
\ELSE 
\STATE \SET $b = b - a$
\ENDIF
\STATE \GOTO \ref{3Start}
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
\caption{}
\algsetup{indent=1.5em}
\begin{algorithmic}[1]
\REQUIRE Reelle Zahl $a\in\mathbb{R}_{\ge 0}$
\ENSURE Reine reelle Zahl $x\in\mathbb{R}$
\STATE \SET $x := 2$ und $y := 1$.
\IF{$|x-y|\le 10^{-10}$} \label{2Start} 
\STATE Gebe $x$ aus.
\RETURN $0$
\ENDIF
\STATE \SET $x := y$
\STATE \SET $y := \frac{1}{2} \cdot \left(x+\frac{a}{x}\right)$
\STATE \GOTO \ref{2Start}
\end{algorithmic}
\end{algorithm}
\end{aufg}
